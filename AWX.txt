INSTALL HELM

curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/main/scripts/get-helm-3
chmod +x get_helm.sh
./get_helm.sh
helm version

ADD HELM REPO

add repo
helm repo add awx-operator https://ansible.github.io/awx-operator/
